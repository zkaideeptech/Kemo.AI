0121: function extractTaggedSection(content: string, tag: string) {
0122:   const pattern = new RegExp(`\\[${tag}\\]([\\s\\S]*?)\\[\\/${tag}\\]`, "i");
0123:   const match = content.match(pattern);
0124:   return match?.[1]?.trim() || "";
0125: }

0127: function parseClarificationItems(block: string) {
0128:   return block
0129:     .split("\n")
0130:     .map((line) => line.trim())
0131:     .filter(Boolean)
0132:     .map((line) => line.replace(/^-+\s*/, ""))
0133:     .map((line) => {
0134:       const match = line.match(/^\u95ee\u9898\uff1a(.+?)(?:\uff5c\u7ebf\u7d22\uff1a(.+))?$/);
0135:       if (!match) {
0136:         return {
0137:           question: line.trim(),
0138:           context: "",
0139:         } satisfies ClarificationQuestion;
0140:       }
0141: 
0142:       return {
0143:         question: match[1].trim(),
0144:         context: (match[2] || "").trim(),
0145:       } satisfies ClarificationQuestion;
0146:     })
0147:     .filter(Boolean) as ClarificationQuestion[];
0148: }
0149: 
0150: function parseLegacyClarificationContent(content: string): ClarificationQuestion[] {
0151:   const segments = content.split(/\u8bf7\u786e\u8ba4[:\uFF1A]/);
0152:   if (segments.length < 2) {
0153:     return [];
0154:   }
0155: 
0156:   const context = segments[0]
0157:     .split("\n")
0158:     .map((line) => line.trim())
0159:     .filter(Boolean)
0160:     .slice(-4)
0161:     .join(" / ");
0162:   const questionLines = segments[1]
0163:     .split("\n")
0164:     .map((line) => line.trim())
0165:     .filter(Boolean)
0166:     .map((line) => line.match(/^\d+[.\u3001]\s*(.+)$/)?.[1]?.trim() || "")
0167:     .filter(Boolean);
0168: 
0169:   return questionLines.map((question) => ({
0170:     question,
0171:     context,
0172:   }));
0173: }

0175: function parseArtifactContent(content: string): ParsedArtifactContent {
0176:   const clarificationBlock = extractTaggedSection(content, "\u5f85\u786e\u8ba4\u9879");
0177:   const bodyBlock = extractTaggedSection(content, "\u8349\u6848\u7248\u6b63\u6587");
0178:   const clarificationItems = clarificationBlock
0179:     ? parseClarificationItems(clarificationBlock)
0180:     : parseLegacyClarificationContent(content);
0181: 
0182:   if (bodyBlock) {
0183:     return {
0184:       body: bodyBlock,
0185:       clarificationItems,
0186:     };
0187:   }
0188: 
0189:   const cleanedContent = content
0190:     .replace(/\[\u5f85\u786e\u8ba4\u9879\][\s\S]*?\[\/\u5f85\u786e\u8ba4\u9879\]/i, "")
0191:     .replace(/\[\u8349\u6848\u7248\u6b63\u6587\]|\[\/\u8349\u6848\u7248\u6b63\u6587\]/gi, "")
0192:     .replace(/\u8bf7\u786e\u8ba4[:\uFF1A][\s\S]*$/i, "")
0193:     .trim();
0194: 
0195:   return {
0196:     body: cleanedContent,
0197:     clarificationItems,
0198:   };
0199: }
